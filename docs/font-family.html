    Font Family

    建议

    参考文档： https://github.com/thx/cube/blob/gh-pages/build/neat.css http://www.uisdc.com/debug-for-smartisan-website 
    
/**
 * 非称线字体
 * 中文优先使用苹方(macOS)、微软雅黑(Windows)
    Helvetica Neue
    Arial
    PingFang SC 苹方
    Microsoft Yahei 微软雅黑
    SimSun 宋体
 */
.sans-serif {
    font-family: "Helvetica Neue", "Arial", "PingFang SC", "Microsoft Yahei", "SimSun", sans-serif;
}


    
/**
 * 称线字体，无宋体系统使用非称线
 */
.serif {
    font-family: "Arial", "PingFang SC", "SimSun", serif;
    *font-family: "SimSun", serif;
}


    
/**
 * 数字字体
 */
.arial {
    font-family: "Arial", sans-serif;
}
